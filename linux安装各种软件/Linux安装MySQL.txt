1、下载 
    wget https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-5.7.22-linux-glibc2.12-x86_64.tar.gz
2、解压
    tar -zxvf  mysql-5.7.22-linux-glibc2.12-x86_64.tar.gz -C /usr/local
3、重命名
    cd /usr/local 
    mv mysql-5.7.22-linux-glibc2.12-x86_64 mysql
4、查看是否安装mysql
    rpm -qa | grep mysql
    安装就删除
    rpm -e --nodeps mysql的版本
5、检查mysql组和用户是否存在，如无创建
    cat /etc/group | grep mysql
    cat /etc/passwd |grep mysql
6、创建用户、用户组
    groupadd mysql
    useradd -r -g mysql mysql //useradd -r参数表示mysql用户是系统用户，不可用于登录系统
7、在mysql下添加data目录
    cd mysql
    mkdir data
8、更改mysql目录下所有的目录及文件夹所属组合用户
    cd /usr/local/
    chown -R mysql mysql/
    chgrp -R mysql mysql/
9、安装mysql
    cd mysql
    如果提示：error while loading shared libraries: libnuma.so.1: cannot open shared object file: No such file or directory
    1.如果已经安装了libnuma.so.1，先yum remove libnuma.so.1
    2.yum -y install numactl.x86_64  
    然后执行：
        ./mysqld --initialize --user=mysql --basedir=/usr/local/mysql/ --datadir=/usr/local/mysql/data/ --lc_messages_dir=/usr/local/mysql/share --lc_messages=en_US    
    保存一下初始密码：>4og=#eR)NkD
10、配置my.cnf
    vim /etc/my.cnf
    加入如下内容：
    #[mysql]  
    #basedir=/usr/local/mysql/  
    #datadir=/usr/local/mysql/data/     
11、配置开机启动 
    cd /usr/local/mysql/support-files/
    cp mysql.server /etc/init.d/mysql
    chmod +x /etc/init.d/mysql 
    chkconfig --add mysql 
12 启动服务
    service mysql start
13、登录
    cd /usr/local/mysql/bin
    ./mysql -u root -p 回车之后输入第九步保存的密码
14、设置密码
    set password=password("root");
15、设置远程登录权限
    grant all privileges on *.* to'root' @'%' identified by 'root';
    FLUSH PRIVILEGES; #刷新权限

修改授权登录方式：（localhost改为任意ip）update user set host='%' where user='root' and host ='localhost';
修改加密规则：ALTER USER 'root'@'%' IDENTIFIED BY 'password' PASSWORD EXPIRE NEVER; 
刷新密码：ALTER USER 'root'@'%' IDENTIFIED WITH mysql_native_password BY 'password'; 
刷新权限：FLUSH PRIVILEGES;
